import simpy
def car(env):  //define una función y se le pasa un argumento
    while True:    
        print('Start parking at %d' %env.now)  enviroment-> objeto de control de eventos, procesos y tiempo. now=tomo el tiempo actual simulación
        parking_duration=5   //Tiempo parkeado 5 unidades de tiempo del simulador
        yield env.timeout(parking_duration)   // yield=generadores + función,proceso,variable =ejecutas comando hasta que env regrese. Yield bloquea el programa hasta que env regrese
// al temrinar el yield :
        print('Start driving at %d' %env.now)  
        trip_duration=2
        yield env.timeout(trip_duration)
